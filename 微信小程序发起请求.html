微信小程序发起请求

Permalink: 1000-01-01 18:44:00+00:09 by 商伟 in 小程序 tags: 小程序

发起请求

发起请求使用的是wx.request相当于axios或者ajax，发起的请求

官方文档 https://developers.weixin.qq.com/miniprogram/dev/api/network/request/wx.request.html

代码
<button bindtap='click'>按钮1</button>  #点击按钮触发click事件

 click:function(){
    var thit = this
    wx.request({
      url: 'http://127.0.0.1:8000/app01/a/',  #访问的路由
      data:{'name':'shang'},   #传参
      method:"POST",    #提交方式
      header:{"content-type":"application/json"},   #请求头header
      success:function(res){  #请求成功的返回值(res)
          console.log(res) 
          console.log(res.data)
          thit.setDate({
              data:res.data
          })
      }
      fail: function (err) { }, #请求失败
      complete: function () { } #请求完成后执行的函数
    })
  }

class A(APIView):
    def post(self,request):
        data = request.data
        return Request(data)
