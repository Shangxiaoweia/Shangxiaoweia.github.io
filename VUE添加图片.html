VUE添加图片

Permalink: 1000-01-01 18:44:00+00:09 by 商伟 in VUE tags: VUE

VUE添加图片

不多说直接上代码

<template>
  <div id="Daa_img">
    <input type="file" id="img">
    <input type="button" @click="add" value="添加">
  </div>
</template>

<script>
  import axios from 'axios'
  export default {
    name: "Daa_img",
    data(){
      return{
        data:'',
      }
    },
    methods:{
      add(){
        let datafrom = new FormData();
        datafrom.append('img',document.getElementById('img').files[0]);
        axios({
          url:'http://127.0.0.1:8000/app1/addimg/',
          method:'post',
          data:datafrom,
        }).then(res=>{
          alert(res)
        })
      }
    }
  }
</script>
