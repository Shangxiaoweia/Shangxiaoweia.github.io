from django.core.paginator import Paginator, EmptyPage, PageNotAnInteger

  Paginator

：创建分页对象

### 分页对象内置属性

all_ = models.objects.all()
p = Paginator(all_, 10)
# 分页all_数据，每页显示10条数据
p.count # 总数据量
p.num_pages() # 分页数
p.page_range() # 列表形式返回当前可有的页数 [1,2,3]
page_1 = p.page(1) # 选择第一页，返回第一页数据对象
