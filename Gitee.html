创建仓库

git config --global user.name "Shangxiaoweia"
git config --global user.email "shangxiaoweia@126.com"

从码云上拿代码

git clone 网址

cd myvue

git pull

git add -A

git commit -m '自定义介绍'

git remote add origin ‘网址’
