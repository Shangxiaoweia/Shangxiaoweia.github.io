Django一对多查询

  • models.py
  from django.db import models

  # 创建用户
  class User(models.Model):
      name = models.CharField(max_length=32)
  # 创建书籍  
  class Book(models.Model):
      title = models.CharField(max_length=32)
      us = models.Foreign('User',on_delete=models.CASCADE)
